⚠️Atualizar Apps Script
Para o sincronismo entre dispositivos funcionar, é necessário atualizar o código do Apps Script para a versão com leitura (GET) e escrita (POST).

Passo 1: Abra sua planilha → Extensões → Apps Script
Passo 2: Substitua todo o código pelo código abaixo:
const SS = SpreadsheetApp.getActiveSpreadsheet(); function doGet(e) { try { const tipo = e.parameter.tipo || 'all'; const result = {}; if (tipo === 'all' || tipo === 'orcamentos') { const sh = SS.getSheetByName('Orcamentos'); result.orcamentos = sh ? sheetToJson(sh) : []; } if (tipo === 'all' || tipo === 'convidados') { const sh = SS.getSheetByName('Convidados'); result.convidados = sh ? sheetToJson(sh) : []; } return ContentService .createTextOutput(JSON.stringify({ok:true, data:result})) .setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService .createTextOutput(JSON.stringify({ok:false,error:err.toString()})) .setMimeType(ContentService.MimeType.JSON); } } function sheetToJson(sh) { const rows = sh.getDataRange().getValues(); if (rows.length < 2) return []; const headers = rows[0]; return rows.slice(1).map(r => { const obj = {}; headers.forEach((h,i) => obj[h] = r[i]); return obj; }); } function doPost(e) { try { const data = JSON.parse(e.postData.contents); if (data.type === 'orcamentos') writeSheet('Orcamentos', data.rows, ['id','desc','cat','valor','status','obs','data']); if (data.type === 'convidados') writeSheet('Convidados', data.rows, ['id','familia','adultos','criancas','status']); return ContentService .createTextOutput(JSON.stringify({ok:true})) .setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService .createTextOutput(JSON.stringify({ok:false,error:err.toString()})) .setMimeType(ContentService.MimeType.JSON); } } function writeSheet(name, rows, fields) { let sh = SS.getSheetByName(name); if (!sh) sh = SS.insertSheet(name); sh.clearContents(); sh.appendRow(fields); rows.forEach(r => sh.appendRow(fields.map(f => r[f] ?? ''))); }
Passo 3: Clique em Implantar → Gerenciar implantações → ✏️ Editar → Nova versão → Implantar
